    <script>

        const pantallaSello = document.getElementById('pantalla-sello');

        const contenedorVideo = document.getElementById('contenedor-video');

        const video = document.getElementById('video-invitacion');

        const invitacion = document.getElementById('invitacion');


        function empezarVideo() {

            // Desvanecer el sello

            pantallaSello.style.opacity = '0';

            

            setTimeout(() => {

                pantallaSello.style.display = 'none';

                // Mostrar y reproducir video

                contenedorVideo.style.display = 'flex';

                video.play();

            }, 800);

        }


        // Cuando el video termina

        video.onended = function() {

            contenedorVideo.style.opacity = '0';

            contenedorVideo.style.transition = 'opacity 1s ease';

            

            setTimeout(() => {

                contenedorVideo.style.display = 'none';

                // Mostrar invitación final

                invitacion.style.display = 'block';

                invitacion.classList.add('reveal');

                window.scrollTo(0, 0);

            }, 1000);

        };
